    <script>
        const stats = {
            p1: { hp: 10000, max: 10000 },
            p2: { hp: 10000, max: 10000 }
        };

        // Generador de Sonidos Sintetizados (Para que no fallen links externos)
        const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

        function playSound(type) {
            const osc = audioCtx.createOscillator();
            const gain = audioCtx.createGain();
            osc.connect(gain);
            gain.connect(audioCtx.destination);

            if (type === 'damage') {
                osc.type = 'sawtooth';
                osc.frequency.setValueAtTime(150, audioCtx.currentTime);
                osc.frequency.exponentialRampToValueAtTime(40, audioCtx.currentTime + 0.2);
                gain.gain.setValueAtTime(0.2, audioCtx.currentTime);
                gain.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.2);
            } else {
                osc.type = 'sine';
                osc.frequency.setValueAtTime(400, audioCtx.currentTime);
                osc.frequency.exponentialRampToValueAtTime(800, audioCtx.currentTime + 0.5);
                gain.gain.setValueAtTime(0.1, audioCtx.currentTime);
                gain.gain.linearRampToValueAtTime(0, audioCtx.currentTime + 0.5);
            }
            osc.start();
            osc.stop(audioCtx.currentTime + 0.5);
        }

        function updateMax(player) {
            const newMax = parseInt(document.getElementById(`max-${player}`).value) || 100;
            stats[player].max = newMax;
            stats[player].hp = newMax; // Reinicia vida al cambiar max
            updateDisplay(player);
        }

        function action(player, type) {
            const card = document.getElementById(`card-${player}`);
            const dmgVal = parseInt(document.getElementById(`dmg-${player}`).value) || 0;
            const healVal = parseInt(document.getElementById(`heal-${player}`).value) || 0;
            
            if (type === 'damage') {
                stats[player].hp -= dmgVal;
                playSound('damage');
                triggerEffect(card, 'aura-damage');
            } else {
                stats[player].hp += healVal;
                playSound('heal');
                triggerEffect(card, 'aura-active');
            }

            if (stats[player].hp < 0) stats[player].hp = 0;
            if (stats[player].hp > stats[player].max) stats[player].hp = stats[player].max;

            updateDisplay(player);
        }

        function triggerEffect(el, cls) {
            el.classList.remove('aura-active', 'aura-damage');
            void el.offsetWidth;
            el.classList.add(cls);
            setTimeout(() => el.classList.remove(cls), 800);
        }

        function updateDisplay(player) {
            const p = stats[player];
            const bar = document.getElementById(`bar-${player}`);
            const text = document.getElementById(`text-${player}`);
            const percent = (p.hp / p.max) * 100;

            bar.style.width = percent + "%";
            text.innerText = `${p.hp} / ${p.max}`;

            if (percent < 25) bar.style.background = "#ff3e3e";
            else if (percent < 50) bar.style.background = "#f1c40f";
            else bar.style.background = "linear-gradient(90deg, #2ecc71, #adff2f)";
        }
    </script>
